# 프로그래머스 Lv.2 [1차] 프렌즈 4블록

## solution 1 빡구현 (오타로 시간 날림 ...)

```python



def solution(m, n, board):
    def OOB(x,y):
        return x+1 >= m or y+1 >= n

        
    _board = [ [e for e in board[i]] for i in range(m)]
    answer = 0
    def isblock(x,y):
        return _board[x][y] == _board[x+1][y] and _board[x+1][y] == _board[x+1][y+1] and _board[x+1][y+1] == _board[x][y+1]
    while True:
        visited = [[False]*n for _ in range(m)]
        for i in range(m):
            for j in range(n):
                if _board[i][j] == '@' : continue
                if OOB(i,j) : 
                    continue
                if isblock(i,j):
                    visited[i][j] = visited[i+1][j] = visited[i+1][j+1] = visited[i][j+1] = True
        for i in range(m):
            for j in range(n):
                if visited[i][j] :
                    _board[i][j] = '@'
        cnt = 0
        for i in range(m):
            for j in range(n):
                if visited[i][j]: cnt+=1
        answer += cnt
        
        if  cnt == 0 : break
        
        for i in range(n):
            for j in range(m-1,-1,-1):
                if _board[j][i] != '@':
                    idx = j+1
                    flag = False
                    while idx < m:
                        if _board[idx][i] != '@':
                            st = _board[j][i]
                            _board[idx-1][i] = st
                            break
                        idx+=1
                        flag = True
                    
                    if idx == m:
                        st = _board[j][i]
                        _board[m-1][i] = st
                    if flag:
                        _board[j][i] = '@'
        
    return answer```
